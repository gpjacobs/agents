def calculate_series(n_terms):
    total = 0
    # Store intermediate results at various points to show convergence
    checkpoints = [10, 100, 1000, 5000, 10000]
    results = {}
    
    for i in range(n_terms):
        denominator = 2 * i + 1  # Generate odd numbers: 1, 3, 5, 7, ...
        term = 1 / denominator
        if i % 2 == 0:  # Even index means positive term
            total += term
        else:  # Odd index means negative term
            total -= term
            
        # Store result at checkpoint
        if i + 1 in checkpoints:
            results[i + 1] = 4 * total
    
    # Multiply by 4 as required for final result
    result = 4 * total
    return result, results

# Calculate the sum of the first 10,000 terms
n_terms = 10000
final_result, intermediate_results = calculate_series(n_terms)

# Print intermediate results to show convergence
print("Convergence pattern:")
for terms, value in intermediate_results.items():
    print(f"After {terms} terms: {value}")

print(f"\nFinal result - Sum of the first {n_terms} terms multiplied by 4: {final_result}")
import math
print(f"Actual value of π: {math.pi}")
print(f"Difference from π: {abs(final_result - math.pi)}")
